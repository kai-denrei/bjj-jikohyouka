```python
#!/usr/bin/env python3
"""
Bank cutover v0.1 → v0.2
Activate ability_axis items, retire all slider10/belt_curve items.
"""
import json
import sys

PILOT_CATS = {'takedowns', 'closed_guard_bottom', 'half_guard_bottom'}

def process_positional(path: str) -> dict:
    with open(path) as f:
        data = json.load(f)

    activated = []
    retired = []
    kept_draft = []

    for q in data['questions']:
        s = q['status']
        inp = q.get('input', '')
        tier = q.get('tier', '')
        cat = q.get('category', '')

        if s == 'draft' and inp == 'ability_axis':
            if tier == 'core' or (tier == 'drilldown' and cat in PILOT_CATS):
                q['status'] = 'active'
                activated.append(q['qid'])
            else:
                kept_draft.append(q['qid'])
        elif s == 'active':
            q['status'] = 'retired'
            retired.append(q['qid'])
        else:
            kept_draft.append(q['qid'])

    with open(path, 'w') as f:
        json.dump(data, f, indent=2, ensure_ascii=False)
        f.write('\n')

    print(f'positional.json: activated={len(activated)}, retired={len(retired)}, kept_draft={len(kept_draft)}')
    print(f'  Activated: {activated}')
    print(f'  Retired count: {len(retired)}')
    print(f'  Kept draft: {kept_draft}')
    return {'activated': activated, 'retired': retired, 'kept_draft': kept_draft}

if __name__ == '__main__':
    result = process_positional('src/data/question-bank/questions/positional.json')
    if len(result['activated']) != 28:
        print(f'ERROR: expected 28 activations, got {len(result["activated"])}', file=sys.stderr)
        sys.exit(1)
    if len(result['retired']) != 173:
        print(f'ERROR: expected 173 retirements, got {len(result["retired"])}', file=sys.stderr)
        sys.exit(1)
    print('Cutover complete.')
```

```bash
import json, os

BANK_DIR = 'src/data/question-bank/questions'
```
